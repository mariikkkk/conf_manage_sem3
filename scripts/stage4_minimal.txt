# 0) ориентация и простой вывод
ls
cd /
ls
ls etc

# 1) cd: относительные и абсолютные пути + подъем вверх
cd etc
ls
cd ..
ls

# 2) head: по умолчанию 10 строк / -n N / альтернативный синтаксис
head readme.txt
head -n 1 readme.txt
head 2 readme.txt

# 3) tac: строки в обратном порядке
tac readme.txt

# 4) du: файл, текущая директория, корень
du readme.txt
du .
du /

# 5) нормализация путей: ., .., лишние слэши
ls .
ls /
ls //etc///
cd /etc/..
ls

# 6) ОБРАБОТКА ОШИБОК (ожидаются сообщения "Ошибка: ...")
# ls по файлу
ls /readme.txt
# cd в несуществующий путь
cd /no_such_path
# head по каталогу
head /etc
# head несуществующего файла
head /nope.txt
# tac по каталогу
tac /etc
# du несуществующего пути
du /nope
